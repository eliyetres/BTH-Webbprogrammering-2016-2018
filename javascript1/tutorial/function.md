Functions
==============================



```
/**
 * Declare a function
 */

function square(number) {
	return number * number;
}



/**
 * Use a function
 */

console.log(square(2));

var x = 3;
var y = square(x);
console.log(y);



/**
 * Declare and use a function with several parameters
 */

function sum(nr1, nr2, nr3) {
	return nr1 + nr2 + nr3;
}

var s = sum(2,4,6);
console.log(s);


```
